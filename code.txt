    const output = document.getElementById('output');

    document.getElementById('voucherForm').onsubmit = async (e) = {
      e.preventDefault();
      const duration = document.getElementById('duration').value;
      const price = document.getElementById('price').value;

      const res = await fetch('http127.0.0.15000create_voucher', {
        method 'POST',
        headers { 'Content-Type' 'applicationjson' },
        body JSON.stringify({ duration_minutes duration, price price })
      });

      const data = await res.json();
      output.innerHTML = `h3Voucher Createdh3div${JSON.stringify(data, null, 2)}div`;
    };

    document.getElementById('loginForm').onsubmit = async (e) = {
      e.preventDefault();
      const username = document.getElementById('loginUser').value;
      const password = document.getElementById('loginPass').value;

      const res = await fetch('http127.0.0.15000login', {
        method 'POST',
        headers { 'Content-Type' 'applicationjson' },
        body JSON.stringify({ username, password })
      });

      const data = await res.json();
      output.innerHTML = `h3Login Responseh3div${JSON.stringify(data, null, 2)}div`;
